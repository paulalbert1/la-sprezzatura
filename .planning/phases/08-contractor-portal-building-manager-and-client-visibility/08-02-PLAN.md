    ```tsx
    <div className="py-2 space-y-2">

        rows={3}
        maxLength={500}

        className="w-full bg-cream-dark text-charcoal border border-stone-light px-4 py-3 font-body text-sm focus:ring-2 focus:ring-terracotta focus:border-terracotta outline-none resize-none"

          <button
            onClick={() => { setText(''); setState('collapsed'); }}
            className="text-xs text-stone hover:text-terracotta underline underline-offset-2 transition-colors"
    ```

    ```tsx
    <p className="text-sm text-stone font-body py-2">Note submitted</p>
    ```

    ```tsx
    <div className="py-2">
    ```

    ```typescript
    const handleSubmit = async () => {
      if (!text.trim()) return;
      setState('submitting');

        setText('');
        setState('success');
        setTimeout(() => setState('collapsed'), 3000);
      } catch {
        setState('error');
      }
    };
    ```
