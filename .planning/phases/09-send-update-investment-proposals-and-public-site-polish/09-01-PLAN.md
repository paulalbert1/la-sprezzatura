**2. Create `src/pages/api/send-update.ts`:**

Follow the `notify-artifact.ts` pattern exactly:
```typescript
export const prerender = false;

import type { APIRoute } from "astro";
import { sanityWriteClient } from "../../sanity/writeClient";
import { generatePortalToken } from "../../lib/generateToken";

export const POST: APIRoute = async ({ request }) => {
  try {
    const { projectId, note, sections } = await request.json();
    if (!projectId) return new Response(JSON.stringify({ error: "projectId required" }), { status: 400, ... });

    // Fetch project snapshot using SEND_UPDATE_PROJECT_QUERY inline (not from queries.ts -- API routes use sanityWriteClient, not sanity:client)
    const project = await sanityWriteClient.fetch(`
      *[_type == "project" && _id == $projectId][0] {
        _id, title, engagementType,
        clients[] { client-> { _id, name, email } },
        milestones[] | order(date asc) { name, date, completed },
        select(engagementType == "full-interior-design" => {
          "procurementItems": procurementItems[] {
            name, status, installDate, retailPrice,
            "savings": retailPrice - clientCost
          }
        }),
        artifacts[] {
          _key, artifactType, customTypeName,
          currentVersionKey,
          "hasApproval": count(decisionLog[action == "approved"]) > 0
        }
      }
    `, { projectId });

    if (!project) return new Response(JSON.stringify({ error: "Project not found" }), { status: 404, ... });

    // Build email HTML
    const baseUrl = import.meta.env.SITE || "https://lasprezz.com";
    const personalNote = note || "";
    const showMilestones = sections?.milestones !== false && project.milestones?.length > 0;
    const showProcurement = sections?.procurement !== false && project.engagementType === "full-interior-design" && project.procurementItems?.length > 0;
    const pendingArtifacts = (project.artifacts || []).filter((a: any) => a.currentVersionKey && !a.hasApproval);
    const showArtifacts = sections?.artifacts !== false && pendingArtifacts.length > 0;

    // Build branded HTML email per 09-UI-SPEC.md template exactly:
    // - Brand header: "LA SPREZZATURA" 11px #8A8478 uppercase
    // - Content card: white bg, 40px 32px padding
    //   - Heading: "Project Update: {title}" 24px Georgia #2C2926 weight 300 centered
    //   - Personal note paragraph: 16px #2C2926 line-height 1.7
    //   - Milestones section: h2 "Milestones" 14px #8A8478 uppercase, table with checkmark/pending rows
    //   - Procurement section: h2 "Procurement" 14px #8A8478 uppercase, table with Item/Status/Install Date headers
    //     - Savings line if total > 0: "You saved {amount} vs. retail" in #059669
    //     - NEVER include clientCost
    //   - Pending reviews: h2 "Items Awaiting Your Review" 14px #8A8478 uppercase, list of artifact labels
    //   - CTA: "View in Your Portal" button bg #C4836A, link to {baseUrl}/portal/dashboard
    // - Footer: "This is a project update from La Sprezzatura." 12px #B8B0A4

    // Use formatCurrency for savings display (import from lib/formatCurrency)
    // Use milestone completed indicator: completed ? "&#10003;" : "&#9675;"
    // Use procurement status inline styles matching STATUS_STYLES colors
    // Date formatting: new Date(date).toLocaleDateString("en-US", { month: "short", day: "numeric", year: "numeric" })

    const html = buildSendUpdateEmail(project, personalNote, showMilestones, showProcurement, showArtifacts, pendingArtifacts, baseUrl);

    // Send to all clients
    const apiKey = import.meta.env.RESEND_API_KEY;
    const recipientEmails: string[] = [];
    if (apiKey) {
      const { Resend } = await import("resend");
      const resend = new Resend(apiKey);
      for (const entry of project.clients || []) {
        const client = entry.client;
        if (!client?.email) continue;
        recipientEmails.push(client.email);
        await resend.emails.send({
          from: "La Sprezzatura <noreply@send.lasprezz.com>",
          to: [client.email],
          subject: `Project Update: ${project.title}`,
          html,
        });
      }
    } else {
      console.log("[SendUpdate] No RESEND_API_KEY set. Would send update for:", projectId);
    }

    // Log the update on the project document
    const sectionsIncluded = [];
    if (showMilestones) sectionsIncluded.push("milestones");
    if (showProcurement) sectionsIncluded.push("procurement");
    if (showArtifacts) sectionsIncluded.push("artifacts");

    await sanityWriteClient
      .patch(projectId)
      .insert("after", "updateLog[-1]", [{
        _key: generatePortalToken(8),
        sentAt: new Date().toISOString(),
        recipientEmails: recipientEmails.join(", "),
        note: personalNote,
        sectionsIncluded,
      }])
      .commit({ autoGenerateArrayKeys: true });

    return new Response(JSON.stringify({ success: true }), { status: 200, headers: { "Content-Type": "application/json" } });
  } catch (error: any) {
    console.error("[SendUpdate] Error:", error);
    return new Response(JSON.stringify({ error: error.message }), { status: 500, headers: { "Content-Type": "application/json" } });
  }
};
```
