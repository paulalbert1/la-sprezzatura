**2. Rewrite Hero.astro per the slideshow design spec:**

This is a full rewrite. Follow `docs/superpowers/specs/2026-03-16-hero-slideshow-design.md` as the authoritative reference.

New Props interface:
```typescript
interface Props {
  images?: Array<{ image: any; alt: string }>;
```

Rendering logic:
- `images` array with 2+ entries: render slideshow
- `images` with exactly 1 entry OR only `image` provided: render single static image
- Neither: render gradient fallback

Inline `<script>` for slideshow + SplitText (per spec):
```typescript
import gsap from "gsap";
import { SplitText } from "gsap/SplitText";
gsap.registerPlugin(SplitText);

let splitInstance: SplitText | null = null;
let slideshowInterval: ReturnType<typeof setInterval> | null = null;
let progressRaf: number | null = null;

document.addEventListener("astro:page-load", () => {
  const heroSection = document.querySelector(".hero");
  if (!heroSection) return;

  const slides = heroSection.querySelectorAll<HTMLElement>(".hero-slide");
  const titleEl = heroSection.querySelector<HTMLElement>(".hero-title");
  const taglineEl = heroSection.querySelector<HTMLElement>(".hero-tagline");
  const progressFill = heroSection.querySelector<HTMLElement>(".hero-progress-fill");
  const prefersReducedMotion = window.matchMedia("(prefers-reduced-motion: reduce)").matches;

  // SplitText animation
  if (titleEl && !prefersReducedMotion) {
    splitInstance = new SplitText(titleEl, { type: "chars" });
    const tl = gsap.timeline();
    tl.fromTo(splitInstance.chars, {
      opacity: 0,
      y: 12,
    }, {
      opacity: 1,
      y: 0,
      duration: 0.06,
      stagger: 0.03,
      ease: "power2.out",
    });
    // Fade in tagline after SplitText completes
    if (taglineEl) {
      tl.to(taglineEl, { opacity: 1, duration: 0.4, ease: "power2.out" });
    }
  } else if (titleEl && prefersReducedMotion) {
    // Show everything immediately
    if (taglineEl) taglineEl.style.opacity = "1";
  }

  // Slideshow
  if (slides.length >= 2 && !prefersReducedMotion) {
    let currentSlide = 0;
    const holdDuration = 7000; // 7s
    const totalDuration = 8500; // 7s hold + 1.5s fade
    let progressStart = Date.now();

    // Ken Burns: apply initial transform and animate
    const kenBurnsDirections: Record<string, { start: string; end: string }> = {
      "push-in": {
        start: "scale(1.0)",
        end: "scale(1.06)",
      },
      "drift-left": {
        start: "translateX(0%) scale(1.0)",
        end: "translateX(-2%) scale(1.06)",
      },
      "pull-back": {
        start: "scale(1.06)",
        end: "scale(1.0)",
      },
      "drift-right": {
        start: "translateX(0%) scale(1.0)",
        end: "translateX(2%) scale(1.06)",
      },
    };

    // Mobile detection for reduced Ken Burns scale
    const isMobile = window.innerWidth < 768;
    if (isMobile) {
      kenBurnsDirections["push-in"].end = "scale(1.03)";
      kenBurnsDirections["drift-left"].end = "translateX(-1%) scale(1.03)";
      kenBurnsDirections["pull-back"].start = "scale(1.03)";
      kenBurnsDirections["drift-right"].end = "translateX(1%) scale(1.03)";
    }

    function applyKenBurns(slide: HTMLElement) {
      const direction = slide.dataset.kenBurnsDirection || "push-in";
      const kb = kenBurnsDirections[direction] || kenBurnsDirections["push-in"];
      const img = slide.querySelector("img");
      if (img) {
        img.style.transition = `transform ${totalDuration}ms ease-out`;
        img.style.transform = kb.start;
        requestAnimationFrame(() => {
          img.style.transform = kb.end;
        });
      }
    }

    // Start Ken Burns on first slide
    applyKenBurns(slides[0]);

    // Progress bar animation
    function updateProgress() {
      if (!progressFill) return;
      const elapsed = Date.now() - progressStart;
      const progress = Math.min(elapsed / holdDuration, 1);
      progressFill.style.width = `${progress * 100}%`;
      if (progress < 1) {
        progressRaf = requestAnimationFrame(updateProgress);
      }
    }
    updateProgress();

    function advanceSlide() {
      const nextSlide = (currentSlide + 1) % slides.length;
      const nextImg = slides[nextSlide].querySelector("img");

      // Image readiness check
      if (nextImg && !nextImg.complete) {
        // Hold current slide, retry in 500ms
        setTimeout(advanceSlide, 500);
        return;
      }

      // Crossfade
      slides[currentSlide].classList.remove("opacity-100");
      slides[currentSlide].classList.add("opacity-0");
      slides[nextSlide].classList.remove("opacity-0");
      slides[nextSlide].classList.add("opacity-100");

      // Ken Burns on new slide
      applyKenBurns(slides[nextSlide]);

      currentSlide = nextSlide;
      progressStart = Date.now();
      if (progressFill) progressFill.style.width = "0%";
      updateProgress();
    }

    slideshowInterval = setInterval(advanceSlide, holdDuration);
  } else if (slides.length >= 2 && prefersReducedMotion) {
    // Show first slide only, no auto-advance, hide progress bar
    const progressBar = heroSection.querySelector(".hero-progress") as HTMLElement;
    if (progressBar) progressBar.style.display = "none";
    if (taglineEl) taglineEl.style.opacity = "1";
  }
});

// Cleanup for view transitions
document.addEventListener("astro:before-swap", () => {
  if (splitInstance) {
    splitInstance.revert();
    splitInstance = null;
  }
  if (slideshowInterval) {
    clearInterval(slideshowInterval);
    slideshowInterval = null;
  }
  if (progressRaf) {
    cancelAnimationFrame(progressRaf);
    progressRaf = null;
  }
});
```
