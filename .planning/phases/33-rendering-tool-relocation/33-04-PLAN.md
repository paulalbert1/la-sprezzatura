**Upload function (port hybrid pattern from Studio, simplified for admin):**
```typescript
async function uploadFile(file: File): Promise<string> {
  if (file.size > 4.5 * 1024 * 1024) {
    // Large file: use @vercel/blob/client with server-issued token
    const { upload } = await import("@vercel/blob/client");
    const blob = await upload(file.name, file, {
      access: "public",
      handleUploadUrl: "/api/blob-upload",
    });
    return blob.pathname;
  }
  // Small file: server-side PUT
  const formData = new FormData();
  formData.append("file", file);
  const res = await fetch("/api/blob-upload", { method: "PUT", body: formData });
  const { pathname } = await res.json();
  return pathname;
}
```
