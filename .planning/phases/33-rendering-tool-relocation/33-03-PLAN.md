**Generate flow (from RESEARCH.md Data Flow):**
```typescript
async function handleGenerate() {
  setIsGenerating(true);
  const res = await fetch("/api/rendering/generate", {
    method: "POST",
    headers: { "Content-Type": "application/json", "x-studio-token": studioToken },
    body: JSON.stringify({
      sessionTitle: wizardData.sessionTitle,
      projectId: wizardData.projectId,
      aspectRatio: wizardData.aspectRatio,
      stylePreset: wizardData.stylePreset,
      description: wizardData.description,
      images: wizardData.images.map(i => ({ blobPathname: i.blobPathname, imageType: i.imageType, location: i.location, notes: i.notes, copyExact: i.copyExact })),
      sanityUserId,
    }),
  });
  const { sessionId } = await res.json();
  // Poll /api/rendering/status every 2s
  const poll = setInterval(async () => {
    const s = await fetch(`/api/rendering/status?sessionId=${sessionId}`, {
      headers: { "x-studio-token": studioToken },
    }).then(r => r.json());
    if (s.status === "complete" || s.status === "error") {
      clearInterval(poll);
      window.location.href = `/admin/rendering/${sessionId}`;
    }
  }, 2000);
}
```
