```typescript
function getBadgeStyle(count: number, limit: number) {
  const pct = limit > 0 ? (count / limit) * 100 : 0;
  if (pct >= 95) return { bg: "#FBEEE8", text: "#9B3A2A", border: "#E8CFA0" };
  if (pct >= 80) return { bg: "#FBF2E2", text: "#8A5E1A", border: "#E8CFA0" };
  return { bg: "#F5EDD8", text: "#9A7B4B", border: "#E8D5A8" };
}
```
