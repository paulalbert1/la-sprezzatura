useEffect(() => {

  const poll = setInterval(() => {
    fetch(`/api/rendering/status?sessionId=${sessionId}`, {
      headers: { "x-studio-token": studioToken },

**Send refinement handler:**
```typescript
async function handleSendRefinement() {
  if (!refinementText.trim() || isRefining) return;
  setIsRefining(true);
  setError(null);
  // Optimistic append (opacity-60)
  const optimisticMsg = { role: "user", text: refinementText, timestamp: new Date().toISOString() };
  setSession(s => s ? { ...s, conversation: [...(s.conversation ?? []), optimisticMsg], status: "generating" } : s);
  setRefinementText("");
  try {
    await fetch("/api/rendering/refine", {
      method: "POST",
      headers: { "Content-Type": "application/json", "x-studio-token": studioToken },
      body: JSON.stringify({ sessionId, refinementText: optimisticMsg.text, sanityUserId }),
    });
    // Poll for completion (handled by the polling effect via status "generating")
  } catch {
    setError("Could not send refinement. Try again.");
    setIsRefining(false);
  }
}
```

**900px breakpoint handling:** Apply `flex-col` and width overrides via a wrapper that checks `useWindowWidth() < 900`. Alternatively, use Tailwind's responsive prefix. Since this is a React island (not pure CSS), use a small inline hook:
```typescript
function useIsNarrow() {
  const [narrow, setNarrow] = useState(false);
  useEffect(() => {
    const mq = window.matchMedia("(max-width: 899px)");
    setNarrow(mq.matches);
    const handler = (e: MediaQueryListEvent) => setNarrow(e.matches);
    mq.addEventListener("change", handler);
    return () => mq.removeEventListener("change", handler);
  }, []);
  return narrow;
}
```
