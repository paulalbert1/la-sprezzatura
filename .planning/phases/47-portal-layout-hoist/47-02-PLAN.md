import { Font } from "astro:assets";
import "../../styles/global.css";
import {
  DEFAULT_PORTAL_BRAND,
  type PortalBrand,
} from "../../lib/portal/portalBrand";
import PortalHeader from "./PortalHeader.astro";
import PortalFooter from "./PortalFooter.astro";
import ImpersonationBannerStub from "./ImpersonationBannerStub.astro";

interface Props {
  title: string;
  /**
   * Optional brand strings sourced from siteSettings via getPortalBrand().
   * Pages that fetch brand data themselves should pass it through; pages
   * that don't fall back to DEFAULT_PORTAL_BRAND so the layout always has
   * something to render.
   */
  tenant?: PortalBrand;
  /**
   * Phase 47 D-2: when true, renders only the wordmark in the header
   * (centered, no role sub-label, no sign-out). Used by /portal/login,
   * /portal/role-select, /workorder/login, /building/login.
   */
  bare?: boolean;
}

const { title, tenant = DEFAULT_PORTAL_BRAND, bare = false } = Astro.props;

// Preview mode: detect ?preview=1 query param. The "Open portal as
// {client}" button on /admin/clients/[clientId] sets this so the
// designer knows they're looking at the client's view. Cosmetic only:
// the param is forgeable (anyone with the URL could append it) and
// has no security implication -- it just renders a banner. The client
// would never legitimately have ?preview=1 in their URL.
//
// Phase 50 will REMOVE this block (CONTEXT D-3) — superseded by the
// real impersonation banner that fills the <slot name="banner"> below.
// Phase 47 leaves it untouched.
const previewMode = Astro.url.searchParams.get("preview") === "1";
---<!-- astro -->

<!doctype html>
<html lang="en">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport" content="width=device-width, initial-scale=1.0" />
    <meta name="robots" content="noindex, nofollow" />
    <Font cssVariable="--font-heading" preload />
    <Font cssVariable="--font-body" preload />
    <title>{title}</title>
  </head>
  <body class="bg-cream text-charcoal font-body min-h-screen flex flex-col">
    {/* Banner slot — Phase 47 D-3. Phase 50 fills with real ImpersonationBanner
        driven by Astro.locals.impersonating. Stub renders an invisible div so
        Plan 01's slot-wiring test passes; empty state has zero rendered height. */}
    <div role="region" aria-label="Notifications">
      <slot name="banner">
        <ImpersonationBannerStub />
      </slot>
    </div>

    {previewMode && (
      <div
        id="preview-banner"
        class="bg-[#9A7B4B] text-white px-6 py-2 text-center text-[12px] tracking-[0.08em] uppercase font-body sticky top-0 z-50"
        role="status"
        aria-live="polite"
      >
        Preview — you are viewing this as a client. Sign back in to <a href="/admin/dashboard" class="underline hover:no-underline">/admin</a> to continue admin work.
      </div>
      <script is:inline>
        // Strip the ?preview=1 param from the visible URL so the flag
        // doesn't leak if the URL is shared. The banner stays for the
        // current page render (server already painted it); subsequent
        // navigations within the portal won't carry the flag.
        try {
          const url = new URL(window.location.href);
          if (url.searchParams.get("preview") === "1") {
            url.searchParams.delete("preview");
            window.history.replaceState({}, "", url.toString());
          }
        } catch {
          /* noop */
        }
      </script>
    )}

    <PortalHeader tenant={tenant} bare={bare} />

    <slot />

    <PortalFooter tenant={tenant} />
  </body>
</html>
