    **2. GSAP ScrollTrigger animations (src/components/animations/ScrollAnimations.astro):**
    Per RESEARCH Pattern 8. Create a self-contained component with inline `<script>`:

    ```
    <script>
      import gsap from "gsap";
      import { ScrollTrigger } from "gsap/ScrollTrigger";
      gsap.registerPlugin(ScrollTrigger);

      function initAnimations() {
        // Fade-in-up for content sections
        gsap.utils.toArray("[data-animate='fade-up']").forEach((el) => {
          gsap.fromTo(el,
            { opacity: 0, y: 24 },
            {
              opacity: 1, y: 0,
              duration: 0.8,
              ease: "power2.out",
              scrollTrigger: {
                trigger: el,
                start: "top 85%",
                toggleActions: "play none none none",
              },
            }
          );
        });

        // Parallax for hero images
        gsap.utils.toArray("[data-animate='parallax']").forEach((el) => {
          gsap.to(el, {
            y: -50,
            ease: "none",
            scrollTrigger: {
              trigger: el,
              start: "top bottom",
              end: "bottom top",
              scrub: true,
            },
          });
        });

        // Staggered reveal for grid items
        gsap.utils.toArray("[data-animate='stagger']").forEach((container) => {
          const items = container.querySelectorAll("[data-animate-item]");
          gsap.fromTo(items,
            { opacity: 0, y: 20 },
            {
              opacity: 1, y: 0,
              duration: 0.6,
              stagger: 0.1,
              ease: "power2.out",
              scrollTrigger: {
                trigger: container,
                start: "top 80%",
                toggleActions: "play none none none",
              },
            }
          );
        });
      }

      function cleanupAnimations() {
        ScrollTrigger.getAll().forEach((trigger) => trigger.kill());
        gsap.killTweensOf("*");
      }

      // Initial load
      document.addEventListener("astro:page-load", () => {
        initAnimations();
      });

      // Cleanup before page swap (view transitions)
      document.addEventListener("astro:before-swap", () => {
        cleanupAnimations();
      });
    </script>
    ```

    Animation types supported:
    - `data-animate="fade-up"`: individual elements fade in and slide up on scroll
    - `data-animate="parallax"`: hero images move slower than scroll for depth
    - `data-animate="stagger"` on container + `data-animate-item` on children: sequential reveal of grid items

    Set initial state in CSS to prevent flash: elements with `[data-animate="fade-up"]` should start with `opacity: 0` via a CSS rule in global.css. Add to global.css:
    ```css
    [data-animate="fade-up"],
    [data-animate-item] {
      opacity: 0;
    }
    ```
