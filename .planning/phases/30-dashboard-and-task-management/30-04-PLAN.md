**Create `src/components/admin/ProjectTasks.tsx`:**

This component is similar to DashboardTasksCard but simplified for a single-project context:
- No project filter dropdown (single project context)
- Full task list (no 5-8 item cap)
- Quick-add form has no project dropdown -- uses the single projectId prop
- Same checkbox toggle with optimistic UI
- Same overdue highlighting with `isTaskOverdue`
- Same error handling with 3-second auto-dismiss

Props:
```typescript
interface TaskItem {
  _key: string;
  description: string;
  dueDate: string | null;
  completed: boolean;
  completedAt: string | null;
  createdAt: string;
}

interface Props {
  tasks: TaskItem[];
  projectId: string;
}
```

State: `localTasks`, `newDescription`, `newDueDate`, `error`, `creating` (same as DashboardTasksCard minus filter-related state).

The toggle and create handlers are identical to DashboardTasksCard except:
- `handleCreate` uses `props.projectId` directly (no project dropdown selection)
- No project title in local task state (single-project context)

Render:
```tsx
<div id="tasks">
  <h2 className="text-sm font-semibold font-body text-charcoal mb-4">Tasks</h2>
  <div className="bg-white rounded-xl border border-stone-light/20 overflow-hidden">
    {/* Task rows -- full list, not capped at 8 */}
    {localTasks.length === 0 ? (
      <p className="text-sm text-stone text-center py-6">No tasks yet. Use the field below to create one.</p>
    ) : (
      <div>
        {localTasks.map((task) => {
          const overdue = isTaskOverdue(task);
          return (
            <div key={task._key} className="flex items-center gap-3 px-5 py-3 border-b border-stone-light/10 last:border-b-0">
              <input
                type="checkbox"
                checked={task.completed}
                onChange={() => handleToggle(task)}
                aria-label={task.description}
                className="w-4 h-4 rounded border-stone-light text-terracotta focus:ring-terracotta accent-terracotta shrink-0"
              />
              <span className={`text-sm font-body flex-1 ${
                task.completed ? "text-stone-light line-through" :
                overdue ? "text-red-600" : "text-charcoal"
              }`}>
                {task.description}
              </span>
              {task.dueDate && (
                <span className={`text-[11px] font-body shrink-0 ${
                  overdue && !task.completed ? "text-red-600 font-medium" : "text-stone-light"
                }`}>
                  {format(parseISO(task.dueDate), "MMM d")}
                </span>
              )}
            </div>
          );
        })}
      </div>
    )}

    {/* Quick-add form -- no project dropdown, pre-scoped to this project */}
    <form onSubmit={handleCreate} className="flex items-center gap-2 px-5 py-3 border-t border-stone-light/10 bg-cream/50">
      <svg className="w-3.5 h-3.5 text-stone-light shrink-0" xmlns="http://www.w3.org/2000/svg" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
        <path d="M5 12h14"/><path d="M12 5v14"/>
      </svg>
      <input
        type="text"
        placeholder="Add a task..."
        value={newDescription}
        onChange={(e) => setNewDescription(e.target.value)}
        disabled={creating}
        className="flex-1 text-sm font-body text-charcoal bg-transparent outline-none placeholder:text-stone-light"
      />
      <input
        type="date"
        value={newDueDate}
        onChange={(e) => setNewDueDate(e.target.value)}
        className="text-xs font-body text-stone bg-transparent border border-stone-light/20 rounded-md px-2 py-1"
      />
    </form>

    {error && (
      <div className="px-5 py-2 text-xs text-red-600 font-body">{error}</div>
    )}
  </div>
</div>
```

The `id="tasks"` on the outer div enables D-14 hash navigation (`#tasks`).

Import `{ useState }` from "react", `{ format, parseISO }` from "date-fns", `{ isTaskOverdue }` from "../../lib/dashboardUtils".

```astro
// -- astro frontmatter open (three dashes) --
export const prerender = false;

import AdminLayout from "../../../../layouts/AdminLayout.astro";
import ProjectTasks from "../../../../components/admin/ProjectTasks.tsx";
import { getAdminProjectDetail } from "../../../../sanity/queries";
import { getTenantClient } from "../../../../lib/tenantClient";
import { STAGE_META } from "../../../../lib/portalStages";
import { getDaysInStage } from "../../../../lib/dashboardUtils";

const projectId = Astro.params.projectId;
if (!projectId) {
  return Astro.redirect("/admin/projects");
}

const tenantId = Astro.locals.tenantId;
if (!tenantId) {
  return Astro.redirect("/admin/login");
}
const client = getTenantClient(tenantId);

const project = await getAdminProjectDetail(client, projectId);
if (!project) {
  return Astro.redirect("/admin/projects");
}

const stageMeta = STAGE_META[project.pipelineStage as keyof typeof STAGE_META];
const daysInStage = getDaysInStage(project.stageChangedAt);

const breadcrumbs = [
  { label: "Dashboard", href: "/admin/dashboard" },
  { label: "Projects", href: "/admin/projects" },
  { label: project.title },
];
// -- astro frontmatter close (three dashes) --

<AdminLayout title={project.title + " - Studio"} pageTitle={project.title} breadcrumbs={breadcrumbs}>
  <div class="max-w-4xl">
    {/* Project header with stage badge and days-in-stage */}
    <div class="flex items-center gap-4 mb-8">
      <h1 class="text-xl font-semibold font-body text-charcoal">{project.title}</h1>
      {stageMeta && (
        <span class="inline-flex px-2.5 py-1 rounded-md text-xs font-medium bg-terracotta/10 text-terracotta">
          {stageMeta.title}
        </span>
      )}
      <span class="text-[11px] text-stone-light font-body">{daysInStage}d in stage</span>
    </div>

    {/* Placeholder sections for future phases (milestones, procurement) */}
    <div id="milestones" class="mb-8">
      {/* Milestones section placeholder -- expanded in future phases */}
    </div>

    <div id="procurement" class="mb-8">
      {/* Procurement section placeholder -- expanded in future phases */}
    </div>

    {/* Tasks section - React island */}
    <ProjectTasks
      client:load
      tasks={project.tasks || []}
      projectId={projectId}
    />
  </div>
</AdminLayout>
```
